  const today = new Date();

  const isCurrentMonth = today.getFullYear() === currentSessionYear && (today.getMonth() + 1) === currentSessionMonth;


  for (let i = startDayOfWeek - 1; i >= 0; i--) {

    const cell = document.createElement('div');

    cell.className = 'calendar-day-cell other-month';

    cell.innerHTML = `<div class="calendar-day-num">${prevMonthTotalDays - i}</div>`;

    grid.appendChild(cell);

  }


  const dayEventsMap = {};

  sessions.forEach(s => {

    const d = new Date(s.date);

    const dayNum = d.getDate();

    if(!dayEventsMap[dayNum]) dayEventsMap[dayNum] = [];

    dayEventsMap[dayNum].push(s);

  });


  for (let day = 1; day <= totalDays; day++) {

    const cell = document.createElement('div');

    cell.className = 'calendar-day-cell';

    

    const currentIterDate = new Date(currentSessionYear, currentSessionMonth - 1, day);

    const dayOfWeek = currentIterDate.getDay();

    

    if (dayOfWeek === 0) cell.classList.add('sun-cell');

    if (dayOfWeek === 6) cell.classList.add('sat-cell');

    if (isCurrentMonth && today.getDate() === day) cell.classList.add('today');


    const monthStr = currentSessionMonth.toString().padStart(2, '0');

    const dayStr = day.toString().padStart(2, '0');

    const staticKey = `${monthStr}-${dayStr}`;

    const dynamicKey = `${currentSessionYear}-${monthStr}-${dayStr}`;

    

    let holidayName = TAIWAN_HOLIDAYS[staticKey] || DYNAMIC_HOLIDAYS_2026[dynamicKey] || '';

    if(holidayName) cell.classList.add('holiday-cell');


    let holidayHTML = holidayName ? `<div class="holiday-label">${holidayName}</div>` : '';


    cell.innerHTML = `

      <div class="calendar-day-num">${day}</div>

      ${holidayHTML}

      <div class="calendar-events-container" id="cal-events-${day}"></div>

    `;


    if(dayEventsMap[day]) {

      cell.classList.add('has-events');

      cell.onclick = function() {

        const firstSessionId = dayEventsMap[day][0].id;

        scrollToSessionCard(firstSessionId);

      };

    }


    grid.appendChild(cell);

  }


  sessions.forEach(s => {

    const d = new Date(s.date);

    const dayNum = d.getDate();

    const container = document.getElementById(`cal-events-${dayNum}`);

    if (container) {

      const ev = document.createElement('div');

      ev.className = 'calendar-event-tag';

      ev.title = `${s.name} (${s.timeStart})`;

      ev.textContent = s.name;

      container.appendChild(ev);

    }

  });


  const currentCells = startDayOfWeek + totalDays;

  const nextMonthCells = currentCells % 7 === 0 ? 0 : 7 - (currentCells % 7);

  for (let i = 1; i <= nextMonthCells; i++) {

    const cell = document.createElement('div');

    cell.className = 'calendar-day-cell other-month';

    cell.innerHTML = `<div class="calendar-day-num">${i}</div>`;

    grid.appendChild(cell);

  }

}


function scrollToSessionCard(sessionId) {

  const card = document.getElementById(`session-card-${sessionId}`);

  if(card) {

    card.scrollIntoView({ behavior: 'smooth', block: 'start' });

    document.querySelectorAll('.session-item').forEach(item => item.classList.remove('highlight-active'));

    card.classList.add('highlight-active');

    setTimeout(() => { card.classList.remove('highlight-active'); }, 3000);

  }

}


function changeSessionMonth(delta) {

  currentSessionMonth += delta;

  if(currentSessionMonth > 12) { currentSessionMonth = 1; currentSessionYear++; }

  if(currentSessionMonth < 1) { currentSessionMonth = 12; currentSessionYear--; }

  renderSessions();

}

function duplicateSession(id) {

  const target = DB.sessions.find(s => s.id === id);

  if(!target) return;


  const d = new Date(target.date);

  d.setDate(d.getDate() + 7);

  const nextWeekDate = d.toISOString().split('T')[0];


  DB.sessions.push({

    id: 's_' + Date.now() + '_' + Math.random().toString(36).substr(2,3),

    name: target.name,

    date: nextWeekDate,

    timeStart: target.timeStart,

    timeEnd: target.timeEnd

  });

  

  saveDB();

  currentSessionYear = d.getFullYear();

  currentSessionMonth = d.getMonth() + 1;

  

  renderSessions();

  showToast(`已複製建立「${target.name}」至 ${nextWeekDate}`);

}

  DB.sessions = DB.sessions.filter(s => s.id !== id);

  saveDB(); renderSessions(); showToast('課程已刪除');

}


/* ── 點名功能 ── */

function getWeekRangeString(dateStr) {

  const d = new Date(dateStr);

  const day = d.getDay(); 

  const sun = new Date(d); sun.setDate(d.getDate() - day);

  const sat = new Date(d); sat.setDate(d.getDate() + (6 - day));

  return {

    key: sun.toISOString().split('T')[0],

    label: `第 ${getWeekNumber(d)} 週 (${sun.getMonth()+1}/${sun.getDate()} ~ ${sat.getMonth()+1}/${sat.getDate()})`

  };

}


function getWeekNumber(d) {

  const target = new Date(d.valueOf());

  const dayNr = (d.getDay() + 6) % 7;

  target.setDate(target.getDate() - dayNr + 3);

  const firstThursday = target.valueOf();

  target.setMonth(0, 1);

  if (target.getDay() !== 4) { target.setMonth(0, 1 + ((4 - target.getDay()) + 7) % 7); }

  return 1 + Math.ceil((firstThursday - target) / 604800000);

}


function renderAttendanceSelector() {

  const filterVal = document.getElementById('att-month-filter').value;

  const [year, month] = filterVal.split('-').map(Number);


  const monthSessions = DB.sessions.filter(s => {

    const d = new Date(s.date);

    return d.getFullYear() === year && (d.getMonth() + 1) === month;

  }).sort((a,b) => new Date(a.date) - new Date(b.date));


  const sel = document.getElementById('att-session-select');

  const previousSelectedValue = sel.value;


  sel.innerHTML = '<option value="">— 請選擇當週要點名的課程 —</option>';


  const groups = {};

  monthSessions.forEach(s => {

    const info = getWeekRangeString(s.date);

    if(!groups[info.key]) groups[info.key] = { label: info.label, sessions: [] };

    groups[info.key].sessions.push(s);

  });


  Object.keys(groups).sort().forEach(weekKey => {

    const optGroup = document.createElement('optgroup');

    optGroup.label = groups[weekKey].label;

    groups[weekKey].sessions.forEach(s => {

      const opt = document.createElement('option');

      opt.value = s.id;

      const isDone = DB.records[s.id] ? '✓ ' : ' ';

      opt.textContent = `${isDone} ${s.date.split('-')[2]}日 - ${s.name}`;

      if(s.id === previousSelectedValue) opt.selected = true;

      optGroup.appendChild(opt);

    });

    sel.appendChild(optGroup);

  });

  

  loadAttendance();

}


function loadAttendance() {

  const sid = document.getElementById('att-session-select').value;

  const area = document.getElementById('att-content');

  if(!sid) { area.style.display = 'none'; return; }

  area.style.display = 'block';


  const session = DB.sessions.find(s => s.id === sid);

  

  const qrEl = document.getElementById('qrcode'); qrEl.innerHTML = '';

  new QRCode(qrEl, { text: `CHECKIN:${sid}`, width: 140, height: 140, colorDark:'#2c2a27', colorLight:'#ffffff' });

  document.getElementById('checkin-code-display').textContent = session.date.replace(/-/g,'').slice(4);


  const rec = DB.records[sid] || {};

  const list = document.getElementById('att-list');

  

  if(!DB.members.length) {

    list.innerHTML = '<div class="empty-state">尚無組員名單，請先至組員管理新增</div>';

    return;

  }


  list.innerHTML = DB.members.map(m => {

    const status = rec[m.id] || '';

    return `

      <div class="att-row">

        <div>

          <div class="att-name">${m.name} <span style="font-size:11px; font-weight:normal; color:var(--text3);">${m.gender || '女'} · ${m.courseType} · ${m.role || '組員'}</span></div>

          <div class="att-id">${m.memberId || ''}</div>

        </div>

        <div class="att-btns">

          <button class="att-btn present ${status==='present'?'active':''}" onclick="setAttendanceState('${sid}','${m.id}','present',this)">出席</button>

          <button class="att-btn leave ${status==='leave'?'active':''}" onclick="setAttendanceState('${sid}','${m.id}','leave',this)">請假</button>

          <button class="att-btn absent ${status==='absent'?'active':''}" onclick="setAttendanceState('${sid}','${m.id}','absent',this)">缺席</button>

        </div>

      </div>

    `;

  }).join('');


  updateAttendanceProgress(sid);

}


function setAttendanceState(sid, mid, status, btn) {

  if(!DB.records[sid]) DB.records[sid] = {};

  if(DB.records[sid][mid] === status) {

    delete DB.records[sid][mid];

    btn.classList.remove('active');

  } else {

    DB.records[sid][mid] = status;

    const row = btn.closest('.att-row');

    row.querySelectorAll('.att-btn').forEach(b => b.classList.remove('active'));

    btn.classList.add('active');

  }

  updateAttendanceProgress(sid);

}


function updateAttendanceProgress(sid) {

  const rec = DB.records[sid] || {};

  const marked = Object.keys(rec).length;

  const total = DB.members.length;

  const pct = total ? Math.round(marked / total * 100) : 0;

  document.getElementById('att-progress-bar').style.width = pct + '%';

  document.getElementById('att-progress-label').textContent = `${marked} / ${total} 人已記錄`;

}

  renderAttendanceSelector();

}

function renderDashboard() {

  document.getElementById('today-date').textContent = new Date().toLocaleDateString('zh-TW', { year:'numeric', month:'long', day:'numeric', weekday:'long' });

  document.getElementById('stat-members').textContent = DB.members.length;

  document.getElementById('stat-sessions').textContent = DB.sessions.length;


  const now = new Date();

  const currentMonthSessions = DB.sessions.filter(s => {

    const d = new Date(s.date);

    return d.getFullYear() === now.getFullYear() && d.getMonth() === now.getMonth();

  });


  let totalPresent = 0, totalPossible = 0, totalAbsent = 0;

  currentMonthSessions.forEach(s => {

    const rec = DB.records[s.id] || {};

    DB.members.forEach(m => {

      totalPossible++;

      if(rec[m.id] === 'present') totalPresent++;

      if(rec[m.id] === 'absent') totalAbsent++;

    });

  });


  document.getElementById('stat-rate').textContent = totalPossible ? Math.round(totalPresent / totalPossible * 100) + '%' : '—';

  document.getElementById('stat-absent').textContent = totalAbsent;

}


function renderStats() {

  const year = parseInt(document.getElementById('stats-year').value);

  const month = parseInt(document.getElementById('stats-month').value);


  const sessions = DB.sessions.filter(s => {

    const d = new Date(s.date);

    if (d.getFullYear() !== year) return false;

    if (month !== 0 && (d.getMonth() + 1) !== month) return false;

    return true;

  }).filter(s => DB.records[s.id]);


  const summaryEl = document.getElementById('stats-summary');

  const tbody = document.getElementById('stats-tbody');


  if(!sessions.length) {

    summaryEl.innerHTML = '';

    tbody.innerHTML = '<tr><td colspan="10" style="text-align:center; color:var(--text3); padding:32px;">指定期間內尚無歷史點名數據</td></tr>';

    return;

  }


  let tP = 0, tA = 0, tL = 0;

  const rows = DB.members.map((m, idx) => {

    let p = 0, a = 0, l = 0;

    sessions.forEach(s => {

      const v = (DB.records[s.id] || {})[m.id];

      if(v === 'present') p++;

      else if(v === 'absent') a++;

      else if(v === 'leave') l++;

    });

    tP += p; tA += a; tL += l;

    const rate = sessions.length ? Math.round(p / sessions.length * 100) : 0;

    return { m, p, a, l, rate, sequence: idx + 1 };

  });


  summaryEl.innerHTML = `

    <div class="stat-card green"><div class="stat-num">${Math.round(tP / (sessions.length * DB.members.length || 1) * 100)}%</div><div class="stat-label">平均出席率</div></div>

    <div class="stat-card"><div class="stat-num">${sessions.length}</div><div class="stat-label">已點名堂數</div></div>

    <div class="stat-card red"><div class="stat-num">${tA}</div><div class="stat-label">累積缺席</div></div>

    <div class="stat-card orange"><div class="stat-num">${tL}</div><div class="stat-label">累積請假</div></div>

  `;


  tbody.innerHTML = rows.map(r => `

    <tr class="clickable-row" onclick="toggleRowHighlight(this)">

      <td class="col-seq"><span style="font-size:13px; color:var(--text3);">${r.sequence}</span></td>

      <td class="col-name"><strong>${r.m.name}</strong></td>

      <td class="col-gender">${r.m.gender || '女'}</td>

      <td class="col-id">${r.m.memberId || '—'}</td>

      <td class="col-type"><span class="course-badge">${r.m.courseType || '神鼓陣'}</span></td>

      <td class="col-role"><span class="role-badge">${r.m.role || '組員'}</span></td>

      <td class="col-num">${r.p}</td>

      <td class="col-num">${r.a}</td>

      <td class="col-num">${r.l}</td>

      <td class="col-rate"><span class="pct ${r.rate>=80?'high':r.rate>=60?'mid':'low'}">${r.rate}%</span></td>

    </tr>

  `).join('');

}


function toggleRowHighlight(rowElement) {

  const isSelected = rowElement.classList.contains('row-selected');

  document.querySelectorAll('#stats-tbody tr').forEach(tr => tr.classList.remove('row-selected'));

  if(!isSelected) {

    rowElement.classList.add('row-selected');

  }

}
